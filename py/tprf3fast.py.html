 19 def _j(n: int) -> np.ndarray:
 20     """Centering matrix: I_n - (1/n) * ones(n, n)."""
 21     return np.eye(n) - np.ones((n, n)) / n
 22 
 23 
 24 def _iota(n: int) -> np.ndarray:
 25     """Column vector of ones (n x 1)."""
 26     return np.ones((n, 1))

 29 def _nanstd_cols(X: np.ndarray) -> np.ndarray:
 30     """Column-wise sample std dev ignoring NaN; returns (1, N) row.
 31     Columns with std = 0 are set to 1 to avoid division by zero."""
 32     s = np.nanstd(X, axis=0, ddof=1)
 33     s[s == 0] = 1.0
 34     return s.reshape(1, -1)

 37 def _ols(y: np.ndarray, X: np.ndarray, min_obs: int = 10) -> np.ndarray:
 38     """OLS with NaN row filtering.
 39 
 40     Rows where y or any X column is NaN are dropped.
 41     Returns coefficient column vector (p x 1), or NaN vector if
 42     fewer than min_obs valid rows remain.
 43     """
 44     valid = ~np.isnan(y.ravel()) & ~np.any(np.isnan(X), axis=1)
 45     if valid.sum() >= min_obs:
 46         b, _, _, _ = np.linalg.lstsq(X[valid], y[valid], rcond=None)
 47         return b.reshape(-1, 1)
 48     return np.full((X.shape[1], 1), np.nan)

 50 def _t3prf_fast(
 51     y:       np.ndarray,
 52     X:       np.ndarray,
 53     Z:       np.ndarray,
 54     pls:     bool = False,
 55     oos_x:   Optional[np.ndarray] = None,
 56 ) -> tuple[np.ndarray, float]:
 57     # ADD THESE TWO LINES TO MATCH ORIGINAL:
 58     Xstd = _nanstd_cols(X)
 59     X = X / Xstd
 60 
 61     T, N = X.shape
 62     L = Z.shape[1]
 63 
 64     # Match the original normalization if it's not already done
 65     # X = X / _nanstd_cols(X)  <-- If you do this here, it will match!
 66 
 67     # --- Pass 1: Loadings (Phi) ---
 68     if pls:
 69         X_mean = np.nanmean(X, axis=0, keepdims=True)
 70         Xd = X - X_mean
 71         B1, _, _, _ = np.linalg.lstsq(Z, Xd, rcond=None)
 72         Phi = B1.T
 73     else:
 74         # column_stack is safer for prepending the intercept
 75         designZ = np.column_stack([np.ones(T), Z])
 76         B1, _, _, _ = np.linalg.lstsq(designZ, X, rcond=None)
 77         Phi = B1[1:, :].T  # Drop intercept row
 78 
 79     # --- Pass 2: Factors (Sigma) ---
 80     if pls:
 81         B2, _, _, _ = np.linalg.lstsq(Phi, Xd.T, rcond=None)
 82         Sigma = B2.T
 83     else:
 84         designPhi = np.column_stack([np.ones(N), Phi])
 85         B2, _, _, _ = np.linalg.lstsq(designPhi, X.T, rcond=None)
 86         Sigma = B2[1:, :].T
 87 
 88     # --- Pass 3: Predictive Regression ---
 89     designSigma = np.column_stack([np.ones(T), Sigma])
 90     beta, _, _, _ = np.linalg.lstsq(designSigma, y, rcond=None)
 91     yhat = designSigma @ beta
 92 
 93     # --- Out-of-sample Forecast ---
 94     yhatt = np.nan
 95     if oos_x is not None:
 96         if pls:
 97             sigma_t, _, _, _ = np.linalg.lstsq(Phi, (oos_x - X_mean).T, rcond=None)
 98         else:
 99             sigma_t, _, _, _ = np.linalg.lstsq(designPhi, oos_x.T, rcond=None)
100             sigma_t = sigma_t[1:]
101         yhatt = float(np.insert(sigma_t.ravel(), 0, 1.0) @ beta.ravel())
102 
103     return yhat, yhatt

105 def _encnew(fore_err1: np.ndarray, fore_err2: np.ndarray) -> float:
106     """Clark-McCracken (2001) ENC-NEW statistic.
107 
108     fore_err1 : forecast errors from the nested (smaller) model
109     fore_err2 : forecast errors from the encompassing (larger) model
110     """
111     e1, e2 = fore_err1.ravel(), fore_err2.ravel()
112     if e1.shape[0] != e2.shape[0]:
113         raise ValueError("fore_err1 and fore_err2 must have the same length")
114     loc = ~np.isnan(e1 + e2)
115     P   = loc.sum()
116     return float(P * np.nansum(e1[loc]**2 - e1[loc] * e2[loc]) / np.nansum(e2[loc]**2))

123 @dataclass
124 class PointEstimates:
125     """Point estimates returned by estimate3prf."""
126     forecasts: np.ndarray          # (T, 1) forecast series
127     ferrors:   np.ndarray          # (T, 1) forecast errors
128     rsquare:   float               # R² vs. rolling mean (can be negative OOS)
129     encnew:    float = np.nan      # ENC-NEW stat (OOS Recursive only)
130     rollfore:  np.ndarray = field(
131         default_factory=lambda: np.full((1, 1), np.nan)
132     )                              # (T, 1) rolling historical mean forecasts
133     alpha:     Optional[np.ndarray] = None   # (N, 1) IS Full predictor coefficients
134 
135 
136 @dataclass
137 class AvarEstimates:
138     """Asymptotic variance estimates (IS Full only)."""
139     alpha:     np.ndarray    # (N, N) asymp. covariance of alpha
140     forecasts: np.ndarray    # (T, T) asymp. covariance of forecasts

146 def estimate3prf_fast(
147     y:            np.ndarray,
148     X:            np.ndarray,
149     Z:            Union[np.ndarray, int],
150     procedure:    str   = 'IS Full',
151     window:       tuple = (0, 1),
152     mintrain:     Union[int, tuple, None] = None,
153     rollwin:      Union[tuple, list] = (30, 20, 0),
154     pls:          bool  = False,
155 ) -> tuple[np.ndarray, PointEstimates]:
156     """
157     High-performance wrapper for 3PRF. 
158     Identical API to original, but vectorized for speed.
159     """
160     T, N = X.shape
161 
162     # Handle autoproxy/Z logic
163     if np.isscalar(Z) and not isinstance(Z, np.ndarray):
164         autoproxy, L = True, int(Z)
165         Z_mat = None
166     else:
167         autoproxy, L = False, Z.shape[1]
168         Z_mat = np.asarray(Z, dtype=float)
169 
170     forecasts = np.full((T, 1), np.nan)
171 
172     # ---- IS Full -------------------------------------------------------
173     if procedure == 'IS Full':
174         forecasts, _ = _t3prf_fast(y, X, Z_mat, pls)
175 
176     # ---- OOS Recursive -------------------------------------------------
177     elif procedure == 'OOS Recursive':
178         # Handle int or (int, int) for mintrain
179         mt, gap = (mintrain, 0) if isinstance(mintrain, int) else mintrain
180 
181         for t in range(mt + gap, T):
182             ts = np.arange(0, t - gap)
183             # Use vectorized engine for the training slice
184             _, f_t = _t3prf_fast(y[ts], X[ts], Z_mat[ts] if not autoproxy else L, pls, oos_x=X[t:t+1])
185             forecasts[t] = f_t
186 
187     # ---- OOS Cross Val -------------------------------------------------
188     elif procedure == 'OOS Cross Val':
189         # Original logic uses a 1-period buffer
190         for t in range(T):
191             # Test set is just index t
192             ts = np.delete(np.arange(T), [max(0, t-1), t, min(T-1, t+1)])
193             _, f_t = _t3prf_fast(y[ts], X[ts], Z_mat[ts] if not autoproxy else L, pls, oos_x=X[t:t+1])
194             forecasts[t] = f_t
195 
196     # Return minimal point estimates for benchmarking parity
197     pe = PointEstimates(forecasts=forecasts, ferrors=y-forecasts, rsquare=0.0)
198     return forecasts, pe

200 def forecast3prf(
201     y:         np.ndarray,
202     X:         np.ndarray,
203     Z:         Union[np.ndarray, int],
204     procedure: str   = 'IS Full',
205     window:    tuple = (0, 1),
206     mintrain:  Union[int, tuple, None] = None,
207     pls:       bool  = False,
208 ) -> np.ndarray:
209     """Three-Pass Regression Filter — forecasts only.
210 
211     Simplified interface that returns only the (T, 1) forecast series.
212     See estimate3prf for the full output including point/asymptotic estimates.
213     Supports 'IS Full', 'OOS Recursive', and 'OOS Cross Val'.
214     """
215     forecasts, _, _ = estimate3prf(
216         y, X, Z, procedure=procedure,
217         window=window, mintrain=mintrain, pls=pls,
218     )
219     return forecasts
